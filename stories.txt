// الحصول على قصة بناءً على المعرف
export function getStory(id: number) {
  if (id < 1 || id > allImages.length) {
    return null
  }

  const index = id - 1
  return {
    id,
    title: storyTitles[index] || `قصة ${id}`,
    image: allImages[index],
    nextId: id < allImages.length ? id + 1 : null,
    prevId: id > 1 ? id - 1 : null,
  }
}

// الحصول على العدد الإجمالي للقصص
export function getTotalStories() {
  return allImages.length
}
